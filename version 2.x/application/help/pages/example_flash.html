Example: Flash

#1 Start a new Actionscript 3 project




#2 Create a document class

package {
	
	import flash.display.Sprite;
	
	public class Main extends Sprite {
		
		public function Main() {
			// Do something
		}
	}
}

#3 Start De MonsterDebugger and export the client class




#4 Include De MonsterDebugger class in your document class

package {
	
	import flash.display.Sprite;
	import nl.demonsters.debugger.MonsterDebugger;
	
	public class Main extends Sprite {
		
		// Variable to hold the debugger
		// This is needed to explore your live application
		private var debugger:MonsterDebugger;
		
		public function Main() {
		
			// Init the debugger
			debugger = new MonsterDebugger(this);
			
			// Send a simple trace
			MonsterDebugger.trace(this, "Hello World!")
		}
	}
}

#5 Run your application
